# Example four --- Stochastic growth model with partial parameterization

states:
k, z
end

jumps:
c, muc
end

shocks:
ϵ
end

parameters:
β 
σ = 1.1
δ = 0.025
α = 0.30
ρ = 0.8
sd = 0.01
end

equations:
k(+1) = (1.0 - δ)*k + exp(z)*k^α - c
c^(-σ) = β*muc(+1)*(1.0 - δ + α*exp(z(+1))*k(+1)^(α - 1.0))
muc = c^(-σ)
z(+1) = ρ*z + sd*ϵ
end
